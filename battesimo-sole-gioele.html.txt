    <!-- Firebase Integration -->
    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
        import { getFirestore, collection, addDoc, onSnapshot } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";
        import { getAuth, signInAnonymously } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js";

        // Inizializzazione icone Lucide
        lucide.createIcons();

        // Configurazione ereditata dal sistema
        const firebaseConfig = JSON.parse('__FIREBASE_CONFIG_JSON__'); 
        const appId = '__APP_ID__';

        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const auth = getAuth(app);

        // Autenticazione Anonima e Monitoraggio Invitati
        signInAnonymously(auth).then(() => {
            const guestsRef = collection(db, 'artifacts', appId, 'public', 'data', 'guests');
            onSnapshot(guestsRef, (snapshot) => {
                const count = snapshot.size;
                document.getElementById('guest-count').innerText = `${count} ${count === 1 ? 'PARTECIPANTE CONFERMATO' : 'PARTECIPANTI CONFERMATI'}`;
            });
        }).catch(err => console.error("Firebase Error:", err));

        // Gestione Invio Modulo
        const form = document.getElementById('rsvp-form');
        form.addEventListener('submit', async (e) => {
            e.preventDefault();
            const btn = document.getElementById('submit-btn');
            btn.disabled = true;
            btn.innerText = "REGISTRAZIONE IN CORSO...";

            try {
                const guestsRef = collection(db, 'artifacts', appId, 'public', 'data', 'guests');
                await addDoc(guestsRef, {
                    firstName: document.getElementById('firstName').value,
                    lastName: document.getElementById('lastName').value,
                    note: document.getElementById('note').value,
                    timestamp: new Date().toISOString()
                });
                
                document.getElementById('rsvp-content').classList.add('hidden');
                document.getElementById('success-msg').classList.remove('hidden');
            } catch (err) {
                console.error("Save Error:", err);
                alert("Si è verificato un errore. Per favore contatta i genitori direttamente.");
                btn.disabled = false;
                btn.innerText = "CONFERMA LA PRESENZA";
            }
        });
    </script>
